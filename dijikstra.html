Dijkstra's Algorithm (C++ Code)

The code below demonstrates the implementation of Dijkstra's algorithm using an adjacency matrix. It calculates the shortest path from a source vertex to all other vertices in a graph.


#include <iostream>
#include <vector>
#include <climits>
using namespace std;

// Function to find the vertex with the smallest distance that hasn't been visited yet
int findMinVertex(vector<int> &dist, vector<bool> &visited, int V) {
    int minVertex = -1;
    for (int i = 0; i < V; i++) {
        if (!visited[i] && (minVertex == -1 || dist[i] < dist[minVertex])) {
            minVertex = i;
        }
    }
    return minVertex;
}

void dijkstra(vector<vector<int>> &graph, int source, int V) {
    vector<int> dist(V, INT_MAX);       // Stores shortest distances from the source
    vector<int> prev(V, -1);           // Stores the parent of each vertex in the shortest path
    vector<bool> visited(V, false);    // Keeps track of visited vertices

    dist[source] = 0; // Distance to the source is 0

    // Main Dijkstra's algorithm loop
    for (int i = 0; i < V - 1; i++) {
        // Find the vertex with the smallest distance
        int u = findMinVertex(dist, visited, V);
        visited[u] = true;

        // Update the distances of adjacent vertices
        for (int v = 0; v < V; v++) {
            if (graph[u][v] != 0 && !visited[v] && dist[u] != INT_MAX && dist[u] + graph[u][v] < dist[v]) {
                dist[v] = dist[u] + graph[u][v];
                prev[v] = u;
            }
        }
    }

    // Print the results
    cout << "Vertex\tDistance from Source\tPenultimate Vertex\n";
    for (int i = 0; i < V; i++) {
        cout << i << "\t" << dist[i] << "\t\t\t" << prev[i] << "\n";
    }
}

int main() {
    int V = 5; // Number of vertices
    vector<vector<int>> graph = {
        {0, 10, 0, 30, 100},
        {0, 0, 50, 0, 0},
        {0, 0, 0, 0, 10},
        {0, 0, 20, 0, 60},
        {0, 0, 0, 0, 0}
    };

    int source = 0; // Starting vertex
    dijkstra(graph, source, V);

    return 0;
}
