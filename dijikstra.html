#include #include #include #include using namespace std; // Define a structure for edges struct Edge { int to, weight; }; // Custom comparator for priority queue struct Compare { bool operator()(pair a, pair b) { return a.second > b.second; // Min-heap based on distance } }; void Dijkstra(vector>& graph, int source) { int V = graph.size(); vector dist(V, INT_MAX); // Distance from source to each vertex vector prev(V, -1); // To store the penultimate vertex vector visited(V, false); // Track visited vertices priority_queue, vector>, Compare> pq; dist[source] = 0; pq.push({source, 0}); // {vertex, distance} while (!pq.empty()) { int u = pq.top().first; // Get the vertex with the smallest distance pq.pop(); if (visited[u]) continue; // Skip if already processed visited[u] = true; // Relaxation of adjacent vertices for (const Edge& edge : graph[u]) { int v = edge.to; int weight = edge.weight; if (!visited[v] && dist[u] + weight < dist[v]) { dist[v] = dist[u] + weight; prev[v] = u; pq.push({v, dist[v]}); } } } // Print shortest distances and paths cout << "Vertex\tDistance from Source\tPenultimate Vertex\n"; for (int i = 0; i < V; ++i) { cout << i << "\t" << dist[i] << "\t\t\t" << prev[i] << "\n"; } } int main() { int V = 5; // Number of vertices vector> graph(V); // Example graph graph[0] = {{1, 10}, {3, 30}, {4, 100}}; graph[1] = {{2, 50}}; graph[2] = {{4, 10}}; graph[3] = {{2, 20}, {4, 60}}; graph[4] = {}; int source = 0; Dijkstra(graph, source); return 0; }
