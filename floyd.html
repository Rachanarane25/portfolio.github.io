C++ Floyd-Warshall Algorithm Code


#include <iostream>
#include <vector>
#include <climits>
using namespace std;

void FloydWarshall(vector<vector<int>>& W) {
    int n = W.size();

    vector<vector<int>>& D = W;

    for (int k = 0; k < n; ++k) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (D[i][k] != INT_MAX && D[k][j] != INT_MAX) {
                    D[i][j] = min(D[i][j], D[i][k] + D[k][j]);
                }
            }
        }
    }

    cout << "Shortest Path Matrix:" << endl;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            if (D[i][j] == INT_MAX) {
                cout << "INF " ;
            } else {
                cout << D[i][j] << " " ;
            }
        }
        cout << endl;
    }
}

int main() {
    int n = 4;
    vector<vector<int>>& W(n, vector<int>(n, INT_MAX));

    for (int i = 0; i < n; ++i) {
        W[i][i] = 0;
    }

    W[0][1] = 5;
    W[0][2] = 10;
    W[1][2] = 3;
    W[2][3] = 1;
    W[3][0] = 7;
    
    FloydWarshall(W);

    return 0;
}
