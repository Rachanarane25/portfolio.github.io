#include #include #include using namespace std; // Floyd's Algorithm for All-Pairs Shortest Path void FloydWarshall(vector>& W) { int n = W.size(); // Initialize the distance matrix D as the weight matrix W vector> D = W; // Apply Floyd-Warshall algorithm for (int k = 0; k < n; ++k) { for (int i = 0; i < n; ++i) { for (int j = 0; j < n; ++j) { // Update the distance matrix with the shortest path if (D[i][k] != INT_MAX && D[k][j] != INT_MAX) { D[i][j] = min(D[i][j], D[i][k] + D[k][j]); } } } } // Print the resulting distance matrix cout << "Shortest Path Matrix:" << endl; for (int i = 0; i < n; ++i) { for (int j = 0; j < n; ++j) { if (D[i][j] == INT_MAX) { cout << "INF "; } else { cout << D[i][j] << " "; } } cout << endl; } } int main() { // Example input: 4 nodes, weights in the adjacency matrix int n = 4; vector> W(n, vector(n, INT_MAX)); // Initializing weight matrix W with some example values // W[i][j] represents the weight of edge from node i to node j for (int i = 0; i < n; ++i) { W[i][i] = 0; // Distance from a node to itself is 0 } // Example edges (node1, node2, weight) W[0][1] = 5; W[0][2] = 10; W[1][2] = 3; W[2][3] = 1; W[3][0] = 7; // Call Floyd's algorithm FloydWarshall(W); return 0; }
