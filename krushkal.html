#include #include #include using namespace std; struct Edge { int u, v, weight; }; class DisjointSet { public: vector parent, rank; DisjointSet(int n) { parent.resize(n); rank.resize(n, 0); for (int i = 0; i < n; ++i) parent[i] = i; } int find(int u) { if (parent[u] != u) parent[u] = find(parent[u]); // Path compression return parent[u]; } void unionSet(int u, int v) { int rootU = find(u); int rootV = find(v); if (rootU != rootV) { // Union by rank if (rank[rootU] > rank[rootV]) parent[rootV] = rootU; else if (rank[rootU] < rank[rootV]) parent[rootU] = rootV; else { parent[rootV] = rootU; rank[rootU]++; } } } }; vector kruskal(int V, vector& edges) { sort(edges.begin(), edges.end(), [](Edge& e1, Edge& e2) { return e1.weight < e2.weight; }); DisjointSet ds(V); vector mst; int totalWeight = 0; // To calculate the sum of MST weights for (const auto& edge : edges) { int u = edge.u, v = edge.v; if (ds.find(u) != ds.find(v)) { mst.push_back(edge); ds.unionSet(u, v); totalWeight += edge.weight; // Add the edge's weight to the total } } // Print the edges in the MST cout << "Edges in the Minimum Spanning Tree (MST):\n"; for (const auto& edge : mst) { cout << edge.u << " - " << edge.v << " : " << edge.weight << endl; } // Output the sum of the weights cout << "Total weight of MST: " << totalWeight << endl; return mst; } int main() { int V = 4; vector edges = { {0, 1, 10}, {0, 2, 6}, {0, 3, 5}, {1, 3, 15}, {2, 3, 4} }; vector mst = kruskal(V, edges); return 0; }
