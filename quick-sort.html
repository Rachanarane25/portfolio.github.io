C++ QuickSort Code


#include <iostream>
#include <vector>
using namespace std;

int Partition(vector<int>& A, int l, int r) {
    int p = A[l];
    int i = l;
    int j = r + 1;

    while (true) {
        do {
            i++;
        } while (A[i] < p && i <= r);

        do {
            j--;
        } while (A[j] > p && j >= l);

        if (i < j) {
            swap(A[i], A[j]);
        } else {
            break;
        }
    }

    swap(A[l], A[j]);
    return j;
}

void QuickSort(vector<int>& A, int l, int r) {
    if (l < r) {
        int s = Partition(A, l, r);
        QuickSort(A, l, s - 1);
        QuickSort(A, s + 1, r);
    }
}

int main() {
    vector<int>& A = {10, 7, 8, 9, 1, 5};
    int n = A.size();

    cout << "Original array: ";
    for (int i = 0; i < n; ++i) {
        cout << A[i] << " ";
    }
    cout << endl;

    QuickSort(A, 0, n - 1);

    cout << "Sorted array: ";
    for (int i = 0; i < n; ++i) {
        cout << A[i] << " ";
    }
    cout << endl;

    return 0;
}
